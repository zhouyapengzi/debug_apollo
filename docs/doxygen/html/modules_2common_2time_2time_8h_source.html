152 // Measure run time of a code block, mostly for debugging purpose.
153 // Example usage:
154 // PERF_BLOCK("Function Foo took: ") {
155 // Foo();
156 // }
157 // You can optionally pass in a time threshold (in second) so that the log will
158 // only be spit out when the elapsed time of running the code block is greater
159 // than it.
160 #define GET_MACRO(_1, _2, NAME, ...) NAME
161 #define PERF_BLOCK(...) \
162  GET_MACRO(__VA_ARGS__, PERF_BLOCK_WITH_THRESHOLD, PERF_BLOCK_NO_THRESHOLD) \
163  (__VA_ARGS__)
164 
165 #define PERF_BLOCK_NO_THRESHOLD(message) PERF_BLOCK_WITH_THRESHOLD(message, 0)
166 
167 #define PERF_BLOCK_WITH_THRESHOLD(message, threshold) \
168  using apollo::common::time::Clock; \
169  for (double block_start_time = 0; \
170  (block_start_time == 0 ? (block_start_time = Clock::NowInSeconds()) \
171  : false); \
172  [&]() { \

174  if (now - block_start_time > (threshold)) { \
175  AINFO << std::fixed << (message) << ": " << now - block_start_time \

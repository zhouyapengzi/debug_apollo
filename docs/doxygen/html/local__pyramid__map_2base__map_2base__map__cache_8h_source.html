116 template <class Key, class Element>
117 Element* LRUCache<Key, Element>::Put(const Key& key, Element* value) {
118  if (value == nullptr) {
119  AINFO << "LRUCache Warning: put a NULL";
120  return nullptr;
121  }
122 
123  Element* node_remove = nullptr;
124  MapIterator found_iter = map_.find(key);
125  if (found_iter != map_.end()) {
126  // key exists in the list
127 
128  // move the corresponding key to list front
129  list_.splice(list_.begin(), list_, found_iter->second);
130 
131  node_remove = found_iter->second->second;
132  if (node_remove == value) {
133  return nullptr;
134  }
135 
136  if (Destroy(node_remove)) {
137  found_iter->second->second = value;
138  } else {
139  node_remove = value;
140  }
141 
142  return node_remove;
143  }
144 
145  // reached capacity, remove value in list, remove key in map
146  if (map_.size() >= capacity_) {
147  node_remove = ClearOne();
148  }
149 
150  if (map_.size() >= capacity_) {
151  AINFO << "LRUCache Warning: the cache size is temporarily increased!";
152  }
153 
154  list_.emplace_front(key, value); // push_front
155  map_[key] = list_.begin();
156 
157  return node_remove;
158 }

160 template <class Key, class Element>
161 Element* LRUCache<Key, Element>::Remove(const Key& key) {
162  Element* node_remove = nullptr;
163  MapIterator found_iter = map_.find(key);
164 
165  if (found_iter != map_.end() && Destroy(found_iter->second->second)) {
166  // try to destruct element
167  node_remove = found_iter->second->second;
168  list_.erase(found_iter->second);
169  map_.erase(found_iter);
170  }
171 
172  return node_remove;
173 }

175 template <class Key, class Element>
176 Element* LRUCache<Key, Element>::ClearOne() {
177  Element* node_remove = nullptr;
178  ListReverseIterator ritr = list_.rbegin();
179  while (ritr != list_.rend()) {
180  if (Destroy(ritr->second)) {
181  node_remove = ritr->second;
182  map_.erase(ritr->first);
183  ritr = ListReverseIterator(list_.erase((++ritr).base()));
184  break;
185  }
186 
187  ++ritr;
188  }
189 
190  return node_remove;
191 }

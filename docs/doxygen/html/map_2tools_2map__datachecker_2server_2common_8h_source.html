110 inline double GetYaw(double from_x, double from_y, double to_x, double to_y) {
111  double vecx = to_x - from_x;
112  double vecy = to_y - from_y;
113  double alpha = acos(vecy / sqrt(vecx * vecx + vecy * vecy));
114  if (vecx < 0) {
115  alpha = 2 * M_PI - alpha;
116  }
117  return kRADIANS_TO_DEGREES * alpha;
118 }
